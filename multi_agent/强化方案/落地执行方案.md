#### 1.2.1 记忆折叠器 (MemoryCompressor)

**文件位置**: `src/memory/compressor.rs`

```rust
/// 记忆胶囊 - 压缩后的记忆单元
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct MemoryCapsule {
    /// 唯一标识
    pub id: String,
    /// 时间范围 [start, end]
    pub time_range: (i64, i64),
    /// 摘要内容
    pub summary: String,
    /// 关键实体提取
    pub entities: Vec<Entity>,
    /// 工具调用记录
    pub tool_calls: Vec<ToolCallSummary>,
    /// 决策点记录
    pub decisions: Vec<Decision>,
    /// 重要性评分 (0.0-1.0)
    pub importance: f32,
    /// 原始对话轮数
    pub original_turns: usize,
    /// 压缩比
    pub compression_ratio: f32,
}

/// 记忆折叠器
pub struct MemoryCompressor {
    /// 触发压缩的上下文阈值 (token 数)
    threshold: usize,
    /// 目标压缩比例
    target_ratio: f32,
    /// 模型 Provider (用于生成摘要)
    provider: Arc<dyn Provider>,
    /// 记忆存储
    memory: Arc<dyn Memory>,
}

impl MemoryCompressor {
    /// 检查是否需要压缩
    pub async fn should_compress(&self, context: &ConversationContext) -> bool {
        context.total_tokens > self.threshold
    }
    
    /// 执行记忆折叠
    pub async fn compress(&self, messages: &[Message]) -> Result<MemoryCapsule> {
        // 1. 提取关键信息
        let entities = self.extract_entities(messages).await?;
        let tool_calls = self.summarize_tool_calls(messages);
        let decisions = self.extract_decisions(messages).await?;
        
        // 2. 生成摘要
        let summary = self.generate_summary(messages).await?;
        
        // 3. 计算重要性
        let importance = self.calculate_importance(&entities, &decisions);
        
        // 4. 构建胶囊
        Ok(MemoryCapsule {
            id: uuid::Uuid::new_v4().to_string(),
            time_range: (messages.first().unwrap().timestamp, messages.last().unwrap().timestamp),
            summary,
            entities,
            tool_calls,
            decisions,
            importance,
            original_turns: messages.len(),
            compression_ratio: 0.0, // 计算后填充
        })
    }
}
```

#### 1.2.2 重要性评分系统

**文件位置**: `src/memory/importance.rs`

```rust
/// 重要性评分因子
#[derive(Debug, Clone)]
pub struct ImportanceFactors {
    /// 用户明确提及的重要性
    pub user_marked: Option<f32>,
    /// 引用次数
    pub reference_count: usize,
    /// 工具调用成功率
    pub tool_success_rate: f32,
    /// 决策影响范围
    pub decision_impact: f32,
    /// 时间衰减因子
    pub time_decay: f32,
}

impl ImportanceScorer {
    /// 计算综合重要性评分
    pub fn score(&self, factors: &ImportanceFactors) -> f32 {
        let weights = &self.config.weights;
        
        let mut score = 0.0;
        score += factors.user_marked.unwrap_or(0.5) * weights.user_marked;
        score += (factors.reference_count as f32).ln().max(0.0) * weights.reference;
        score += factors.tool_success_rate * weights.tool_success;
        score += factors.decision_impact * weights.decision;
        score *= factors.time_decay;
        
        score.clamp(0.0, 1.0)
    }
    
    /// 时间衰减函数
    pub fn time_decay(&self, age_hours: f64, half_life_hours: f64) -> f32 {
        2.0_f32.powf(-(age_hours / half_life_hours) as f32)
    }
}
```

```rust
/// 上下文管理配置
#[derive(Debug, Clone, Deserialize)]
pub struct ContextManagerConfig {
    /// 最大上下文 token 数
    pub max_context_tokens: usize,
    /// 触发压缩的阈值比例
    pub compression_threshold_ratio: f32,
    /// 滑动窗口大小（最近 N 轮）
    pub sliding_window_size: usize,
    /// 关键事件缓存大小
    pub key_event_cache_size: usize,
    /// 重要记忆召回数量
    pub important_memory_recall: usize,
}

/// 自动上下文管理器
pub struct ContextManager {
    config: ContextManagerConfig,
    compressor: Arc<MemoryCompressor>,
    memory: Arc<dyn Memory>,
    importance_scorer: Arc<ImportanceScorer>,
}

impl ContextManager {
    /// 构建当前上下文
    pub async fn build_context(&self, agent_id: &str, current_message: &Message) -> Result<Vec<Message>> {
        let mut context = Vec::new();
        
        // 1. 系统提示（始终包含）
        context.push(self.get_system_prompt());
        
        // 2. 关键记忆胶囊（按重要性召回）
        let capsules = self.recall_important_memories(agent_id).await?;
        for capsule in capsules {
            context.push(Message::system(&capsule.to_prompt()));
        }
        
        // 3. 最近对话（滑动窗口）
        let recent = self.get_recent_conversation(agent_id).await?;
        context.extend(recent);
        
        // 4. 当前消息
        context.push(current_message.clone());
        
        // 5. 检查是否需要压缩
        if self.estimate_tokens(&context) > self.config.max_context_tokens {
            context = self.compress_and_rebuild(agent_id, context).await?;
        }
        
        Ok(context)
    }
    
    /// 召回重要记忆
    async fn recall_important_memories(&self, agent_id: &str) -> Result<Vec<MemoryCapsule>> {
        self.memory.search_memories(SearchQuery {
            agent_id: Some(agent_id.to_string()),
            memory_type: MemoryType::Capsule,
            limit: self.config.important_memory_recall,
            sort_by: SortBy::Importance,
        }).await
    }
}
```

```rust
    /// 最大内存 (MB)
    pub max_memory_mb: usize,
    /// 最大执行时间 (ms)
    pub max_exec_time_ms: u64,
    /// 最大 CPU 使用率

    /// 是否允许文件系统访问

impl WasmSkillRuntime {
    /// 加载 WASM 技能
    pub async fn load_skill(&self, wasm_bytes: &[u8]) -> Result<WasmSkillInstance> {
        // 1. 验证 WASM 模块
        self.validate_module(wasm_bytes)?;
        
        // 2. 编译模块
        let module = wasmtime::Module::new(&self.engine, wasm_bytes)?;
        
        // 3. 配置限制
        let mut config = wasmtime::Config::new();
        config.set_max_memory(self.limits.max_memory_mb * 1024 * 1024);
        
        // 4. 创建实例
        Ok(WasmSkillInstance {
            module,
            limits: self.limits.clone(),
        })
    }
    
    /// 执行技能
    pub async fn execute(
        &self,
        instance: &mut WasmSkillInstance,
        input: SkillInput,
    ) -> Result<SkillOutput> {
        // 设置超时
        let result = tokio::time::timeout(
            Duration::from_millis(self.limits.max_exec_time_ms),
            self.execute_internal(instance, input),
        ).await??;
        
        Ok(result)
    }
}
```

```rust
/// 依赖解析器
pub struct DependencyResolver {
    repositories: Vec<Arc<dyn SkillRepository>>,
    cache: Arc<SkillCache>,
}

impl DependencyResolver {
    /// 解析依赖树
    pub async fn resolve(&self, skill_id: &str, version: Option<&str>) -> Result<DependencyTree> {
        let mut tree = DependencyTree::new();
        self.resolve_recursive(skill_id, version, &mut tree, HashSet::new()).await?;
        Ok(tree)
    }
    
    async fn resolve_recursive(
        &self,
        skill_id: &str,
        version: Option<&str>,
        tree: &mut DependencyTree,
        visited: &mut HashSet<String>,
    ) -> Result<()> {
        let key = format!("{}@{}", skill_id, version.unwrap_or("latest"));
        if visited.contains(&key) {
            return Ok(()); // 已处理，跳过
        }
        visited.insert(key.clone());
        
        // 获取技能
        let manifest = self.get_skill_manifest(skill_id, version).await?;
        
        // 递归解析依赖
        for dep in &manifest.dependencies {
            self.resolve_recursive(&dep.id, Some(&dep.version), tree, visited).await?;
        }
        
        tree.add_skill(skill_id, version, manifest);
        Ok(())
    }
}
```

```rust
pub struct ExperienceCapsule {

    pub id: String,
    /// 来源 Agent

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct StrategyTemplate {
    /// 步骤序列
    pub steps: Vec<StrategyStep>,
    /// 工具调用序列
    pub tool_sequence: Vec<ToolInvocation>,
    /// 决策点
    pub decision_points: Vec<DecisionPoint>,

pub struct ExperienceExtractor {
    /// 模型 Provider
    provider: Arc<dyn Provider>,
    /// 经验存储
    experience_store: Arc<dyn ExperienceStore>,
    /// 最小样本数（生成经验前）
    min_samples: usize,
}

impl ExperienceExtractor {
    /// 从执行轨迹提取经验
    pub async fn extract(&self, trace: &ExecutionTrace) -> Result<Option<ExperienceCapsule>> {
        // 1. 分析执行结果
        if !self.is_successful(trace) {
            // 失败的执行也可能有价值
            return self.extract_failure_lesson(trace).await;
        }
        
        // 2. 提取关键步骤
        let key_steps = self.identify_key_steps(trace).await?;
        
        // 3. 识别触发条件
        let conditions = self.extract_conditions(trace).await?;
        
        // 4. 生成策略模板
        let strategy = self.generate_strategy(trace, &key_steps).await?;
        
        // 5. 计算置信度
        let confidence = self.calculate_confidence(trace);
        
        if confidence < 0.5 {
            return Ok(None); // 置信度过低，不生成经验
        }
        
        Ok(Some(ExperienceCapsule {
            id: uuid::Uuid::new_v4().to_string(),
            source_agent: trace.agent_id.clone(),
            task_type: trace.task_type.clone(),
            strategy,
            trigger_conditions: conditions,
            expected_outcome: trace.goal.clone(),
            actual_outcome: Outcome::Success,
            confidence,
            usage_count: 0,
            success_count: 0,
            created_at: chrono::Utc::now().timestamp(),
            last_used_at: 0,
        }))
    }
    
    /// 提取失败教训
    async fn extract_failure_lesson(&self, trace: &ExecutionTrace) -> Result<Option<ExperienceCapsule>> {
        // 分析失败原因，生成避免策略
        let failure_analysis = self.analyze_failure(trace).await?;

            id: uuid::Uuid::new_v4().to_string(),

            task_type: trace.task_type.clone(),
            strategy: StrategyTemplate {
                steps: vec![],
                tool_sequence: vec![],
                decision_points: vec![],
                fallback: None,
            },
            trigger_conditions: failure_analysis.warning_signs,
            expected_outcome: "Avoid this failure pattern".to_string(),
            actual_outcome: Outcome::Failure(failure_analysis.reason),
            confidence: 0.7,
            usage_count: 0,
            success_count: 0,
            created_at: chrono::Utc::now().timestamp(),
            last_used_at: 0,
        }))
    }
}
```

```rust
/// 经验池 - 团队/集群级别共享
pub struct ExperiencePool {
    /// 存储后端
    store: Arc<dyn ExperienceStore>,
    /// 订阅者
    subscribers: DashMap<String, Vec<String>>,
    /// 所属级别
    level: MemoryLevel,
    /// 所属团队/实例
    scope_id: String,
}

#[async_trait]
pub trait ExperienceStore: Send + Sync {
    /// 存储经验
    async fn store(&self, capsule: &ExperienceCapsule) -> Result<()>;
    
    /// 检索相关经验
    async fn search(&self, query: &ExperienceQuery) -> Result<Vec<ExperienceCapsule>>;
    
    /// 更新使用统计
    async fn update_stats(&self, id: &str, success: bool) -> Result<()>;
    
    /// 获取高置信度经验
    async fn get_top_experiences(&self, task_type: &TaskType, limit: usize) -> Result<Vec<ExperienceCapsule>>;
}

impl ExperiencePool {
    /// 发布经验到池中
    pub async fn publish(&self, capsule: ExperienceCapsule) -> Result<()> {
        // 1. 存储到本地
        self.store.store(&capsule).await?;
        
        // 2. 通知订阅者
        self.notify_subscribers(&capsule).await?;
        
        // 3. 如果是团队级别，传播到集群级别
        if self.level == MemoryLevel::Team {
            self.propagate_to_cluster(&capsule).await?;
        }
        
        Ok(())
    }
    
    /// 订阅相关经验
    pub async fn subscribe(&self, agent_id: &str, task_types: &[TaskType]) -> Result<()> {
        for task_type in task_types {
            self.subscribers
                .entry(task_type.to_string())
                .or_default()
                .push(agent_id.to_string());
        }
        Ok(())
    }
    
    /// 获取相关经验
    pub async fn get_relevant(&self, task_type: &TaskType, context: &str) -> Result<Vec<ExperienceCapsule>> {
        // 1. 按任务类型搜索
        let mut results = self.store.search(&ExperienceQuery {
            task_type: Some(task_type.clone()),
            min_confidence: 0.6,
            limit: 10,
        }).await?;
        
        // 2. 按上下文相似度排序
        results.sort_by(|a, b| {
            let sim_a = self.context_similarity(context, &a.trigger_conditions);
            let sim_b = self.context_similarity(context, &b.trigger_conditions);
            sim_b.partial_cmp(&sim_a).unwrap()
        });
        
        Ok(results)
    }
}
```

   - 基于 hint 的智能路由
   - 规则可配置

4. **Agent 指定模型**
   - 通过配置指定每个 Agent 使用的模型
   - 支持模型级别的 fallback

### 4.2 使用示例

```rust
// 配置文件 config.toml
[providers.openrouter]
type = "openrouter"
api_key = "${OPENROUTER_API_KEY}"

[providers.alibaba]
type = "qwen"
api_key = "${DASHSCOPE_API_KEY}"
region = "cn"

[router]
default = "openrouter"
routes = [
    { hint = "reasoning", provider = "openrouter", model = "anthropic/claude-3-opus" },
    { hint = "fast", provider = "alibaba", model = "qwen-turbo" },
    { hint = "code", provider = "openrouter", model = "deepseek/deepseek-coder" },
]

// Agent 配置指定模型
[[agents.researcher]]
provider = "openrouter"
model = "openrouter/auto"  # 自动选择

[[agents.coder]]
provider = "alibaba"
model = "qwen-coder-plus"
```

此目标已完成，只需更新文档说明使用方式。
